151template <typename Func>
152int NewtonTrustEig(Func costFunc, Eigen::VectorXd & x, Eigen::VectorXd & g, Eigen::MatrixXd & Q, Eigen::VectorXd & v, int verbosity = 0)
153{
154 typedef double Scalar;
155 typedef Eigen::VectorXd Vector;
156 typedef Eigen::MatrixXd Matrix;
157
158 assert(x.cols() == 1);
159 g.resize(x.size());
160 Q.resize(x.size(), x.size());
161 v.resize(x.size());
162
163 Matrix H(x.size(), x.size());
164
165 // Storage for trial point, gradient and Hessian
166 Vector xn(x.size());
167 Vector gn(x.size());
168 Matrix Hn(x.size(), x.size());
169
170 // Evaluate initial cost, gradient and Hessian
171 Scalar f = costFunc(x, g, H);
172 if (!std::isfinite(f) || !g.allFinite() || !H.allFinite()) // if any nan, -inf or +inf
173 {
174 if (verbosity > 1)
175 std::println("ERROR: Initial point is not in domain of cost function");
176 return -1;
177 }
178
179 // Eigendecomposition of initial Hessian
180 Eigen::SelfAdjointEigenSolver<Matrix> eigenH(H);
181 v = eigenH.eigenvalues();
182 Q = eigenH.eigenvectors();
183
184 Scalar Delta = 1e0; // Initial trust-region radius
185
186 const int maxIterations = 5000;
187 for (int i = 0; i < maxIterations; ++i)
188 {
189 // Solve trust-region subproblem
190 Vector p;
191 trsEig(Q, v, g, Delta, p); // minimise 0.5*p.'*H*p + g.'*p subject to ||p|| <= Delta
192
193 Scalar pg = p.dot(g);
194 Scalar LambdaSq = -pg; // The Newton decrement squared is g.'*inv(H)*g = p.'*H*p
195 if (verbosity == 3)
196 std::println("Iter = {:5}, Cost = {:10.2e}, Newton decr^2 = {:10.2e}, Delta = {:10.2e}", i, f, LambdaSq, Delta);;
197 if (verbosity == 1)
198 std::print(".");
199
200 // const Scalar LambdaSqThreshold = std::sqrt(std::numeric_limits<Scalar>::epsilon()); // Loose convergence tolerance
201 const Scalar LambdaSqThreshold = 2*std::numeric_limits<Scalar>::epsilon(); // Tight convergence tolerance
202 if (std::fabs(LambdaSq) < LambdaSqThreshold)
203 {
204 if (verbosity >= 2)
205 std::println("CONVERGED: Newton decrement below threshold in {} iterations", i);
206 return 0;
207 }
208
209 // Evaluate cost, gradient and Hessian for trial step
210 xn = x + p;
211 Scalar fn = costFunc(xn, gn, Hn);
212 bool outOfDomain = !std::isfinite(fn) || !gn.allFinite() || !Hn.allFinite(); // if any nan, -inf or +inf
213 Scalar rho;
214 if (outOfDomain)
215 {
216 rho = -1; // Force trust region reduction and reject step
217 }
218 else
219 {
220 Scalar dm = -pg - 0.5*p.dot(H*p); // Predicted reduction f - fp, where fp = f + p'*g + 0.5*p'*H*p
221 rho = (f - fn)/dm; // Actual reduction divided by predicted reduction
222 }
223
224 if (rho < 0.1)
225 {
226 Delta = 0.25*p.norm(); // Decrease trust region radius
227 }
228 else
229 {
230 if (rho > 0.75 && p.norm() > 0.8*Delta)
231 {
232 Delta = 2.0*Delta; // Increase trust region radius
233 }
234 }
235
236 if (rho >= 0.001)
237 {
238 // Accept the step
239 x = xn;
240 f = fn;
241 g = gn;
242 H = Hn;
243
244 // Eigendecomposition of accepted Hessian
245 Eigen::SelfAdjointEigenSolver<Matrix> eigenH_(H);
246 v = eigenH_.eigenvalues();
247 Q = eigenH_.eigenvectors();
248 }
249 }
250 if (verbosity > 1)
251 std::println("WARNING: maximum number of iterations reached");
252 return 1;
253}

344template <typename Func>
345int SR1TrustEig(Func costFunc, Eigen::VectorXd & x, Eigen::VectorXd & g, Eigen::MatrixXd & Q, Eigen::VectorXd & v, int verbosity = 0)
346{
347 typedef double Scalar;
348 typedef Eigen::VectorXd Vector;
349 typedef Eigen::MatrixXd Matrix;
350
351 assert(x.cols() == 1);
352 g.resize(x.size());
353 Q.resize(x.size(), x.size());
354 v.resize(x.size());
355
356 Matrix H(x.size(),x.size());
357
358 // Storage for trial point and gradient
359 Vector xn(x.size());
360 Vector gn(x.size());
361
362 // Evaluate initial cost and gradient
363 Scalar f = costFunc(x, g);
364 if (!std::isfinite(f) || !g.allFinite()) // if any nan, -inf or +inf
365 {
366 if (verbosity > 1)
367 std::println("ERROR: Initial point is not in domain of cost function");
368 return -1;
369 }
370
371 // Initial Hessian
372 H = Q*v.asDiagonal()*Q.transpose();
373
374 Scalar Delta = 1e0; // Initial trust-region radius
375
376 const int maxIterations = 5000;
377 for (int i = 0; i < maxIterations; ++i)
378 {
379 // Solve trust-region subproblem
380 Vector p;
381 trsEig(Q, v, g, Delta, p); // minimise 0.5*p.'*H*p + g.'*p subject to ||p|| <= Delta
382
383 Scalar pg = p.dot(g);
384 Scalar LambdaSq = -pg; // The Newton decrement squared is g.'*inv(H)*g = p.'*H*p
385 if (verbosity == 3)
386 std::println("Iter = {:5}, Cost = {:10.2e}, Newton decr^2 = {:10.2e}, Delta = {:10.2e}", i, f, LambdaSq, Delta);;
387 if (verbosity == 1)
388 std::print(".");
389
390 const Scalar LambdaSqThreshold = std::sqrt(std::numeric_limits<Scalar>::epsilon()); // Loose convergence tolerance
391 // const Scalar LambdaSqThreshold = 2*std::numeric_limits<Scalar>::epsilon(); // Tight convergence tolerance
392 if (std::fabs(LambdaSq) < LambdaSqThreshold && v(0) > 0.0)
393 {
394 if (verbosity >= 2)
395 std::println("CONVERGED: Newton decrement below threshold in {} iterations", i);
396 return 0;
397 }
398
399 // Evaluate cost and gradient for trial step
400 xn = x + p;
401 Scalar fn = costFunc(xn, gn);
402 bool outOfDomain = !std::isfinite(fn) || !gn.allFinite(); // if any nan, -inf or +inf
403 Vector y = gn - g;
404
405 Scalar rho;
406 if (outOfDomain)
407 {
408 rho = -1; // Force trust region reduction and reject step
409 }
410 else
411 {
412 Scalar dm = -pg - 0.5*p.dot(H*p); // Predicted reduction f - fp, where fp = f + p'*g + 0.5*p'*H*p
413 rho = (f - fn)/dm; // Actual reduction divided by predicted reduction
414 }
415
416 if (rho < 0.1)
417 {
418 Delta = 0.25*p.norm(); // Decrease trust region radius
419 }
420 else
421 {
422 if (rho > 0.75 && p.norm() > 0.8*Delta)
423 {
424 Delta = 2.0*Delta; // Increase trust region radius
425 }
426 }
427
428 if (rho >= 0.001)
429 {
430 // Accept the step
431 x = xn;
432 f = fn;
433 g = gn;
434 }
435
436 // Update Hessian approximation
437 const Scalar sqrteps = std::sqrt(std::numeric_limits<Scalar>::epsilon());
438 if (!outOfDomain)
439 {
440 Vector w = y - H*p;
441 Scalar pw = p.dot(w);
442 if (std::fabs(pw) > sqrteps*p.norm()*w.norm())
443 {
444 Scalar s = sgn(pw);
445 Vector u = w.array()/std::sqrt(s*pw);
446 H = H + s*(u*u.transpose());
447
448 // Eigendecomposition of updated Hessian
449 Eigen::SelfAdjointEigenSolver<Matrix> eigenH(H);
450 v = eigenH.eigenvalues();
451 Q = eigenH.eigenvectors();
452 // We should do a rank-one update of Q and v instead of a full eigendecomposition using, e.g.,
453 // [1] Bunch, J.R., Nielsen, C.P. and Sorensen, D.C., 1978. Rank-one modification of the symmetric eigenproblem. Numerische Mathematik, 31(1), pp.31-48.
454 }
455 }
456 }
457 if (verbosity > 1)
458 std::println("WARNING: maximum number of iterations reached");
459 return 1;
460}

542template <typename Func>
543int BFGSTrustSqrt(Func costFunc, Eigen::VectorXd & x, Eigen::VectorXd & g, Eigen::MatrixXd & Xi, int verbosity = 0)
544{
545 typedef double Scalar;
546 typedef Eigen::VectorXd Vector;
547 typedef Eigen::MatrixXd Matrix;
548
549 assert(x.cols() == 1);
550 g.resize(x.size());
551 assert(Xi.isUpperTriangular());
552
553 // Storage for trial point and gradient
554 Vector xn(x.size());
555 Vector gn(x.size());
556
557 // Evaluate initial cost and gradient
558 Scalar f = costFunc(x, g);
559 if (!std::isfinite(f) || !g.allFinite()) // if any nan, -inf or +inf
560 {
561 if (verbosity > 1)
562 std::println("ERROR: Initial point is not in domain of cost function");
563 return -1;
564 }
565
566 Scalar Delta = 10e0; // Initial trust-region radius
567
568 const int maxIterations = 5000;
569 for (int i = 0; i < maxIterations; ++i)
570 {
571 // Solve trust-region subproblem
572 Vector p;
573 trsSqrt(Xi, g, Delta, p); // minimise 0.5*p.'*Xi.'*Xi*p + g.'*p subject to ||Xi*p|| <= Delta
574 Vector z = Xi*p;
575
576 Scalar pg = p.dot(g);
577 Scalar LambdaSq = -pg; // The Newton decrement squared is g.'*inv(H)*g = p.'*H*p = p.'*Xi.'*Xi*p
578 if (verbosity == 3)
579 std::println("Iter = {:5}, Cost = {:10.2e}, Newton decr^2 = {:10.2e}, Delta = {:10.2e}", i, f, LambdaSq, Delta);;
580 if (verbosity == 1)
581 std::print(".");
582
583 // const Scalar LambdaSqThreshold = std::sqrt(std::numeric_limits<Scalar>::epsilon()); // Loose convergence tolerance
584 const Scalar LambdaSqThreshold = 2*std::numeric_limits<Scalar>::epsilon(); // Tight convergence tolerance
585 if (std::fabs(LambdaSq) < LambdaSqThreshold)
586 {
587 if (verbosity >= 2)
588 std::println("CONVERGED: Newton decrement below threshold in {} iterations", i);
589 return 0;
590 }
591
592 // Evaluate cost and gradient for trial step
593 xn = x + p;
594 Scalar fn = costFunc(xn, gn);
595 bool outOfDomain = !std::isfinite(fn) || !gn.allFinite(); // if any nan, -inf or +inf
596 Vector y = gn - g;
597
598 Scalar rho;
599 if (outOfDomain)
600 {
601 rho = -1; // Force trust region reduction and reject step
602 }
603 else
604 {
605 Scalar dm = -pg - 0.5*z.squaredNorm(); // Predicted reduction f - fp, where fp = f + p'*g + 0.5*p'*H*p
606 rho = (f - fn)/dm; // Actual reduction divided by predicted reduction
607 }
608
609 if (rho < 0.1)
610 {
611 Delta = 0.25*z.norm(); // Decrease trust region radius
612 }
613 else
614 {
615 if (rho > 0.75 && z.norm() > 0.8*Delta)
616 {
617 Delta = 2.0*Delta; // Increase trust region radius
618 }
619 }
620
621 if (rho >= 0.001)
622 {
623 // Accept the step
624 x = xn;
625 f = fn;
626 g = gn;
627 }
628
629 // Update Hessian approximation
630 const Scalar sqrteps = std::sqrt(std::numeric_limits<Scalar>::epsilon());
631 if (!outOfDomain)
632 {
633 Scalar py = p.dot(y);
634 if (py > sqrteps*y.norm()*p.norm())
635 {
636 // Form
637 // [ Xi*p, Xi ]
638 // [ 0, y^T/sqrt(y^T*p) ]
639 Matrix RR(Xi.rows() + 1, Xi.cols() + 1);
640 RR << Xi*p, Xi,
641 0, y.transpose()/std::sqrt(py);
642 // Q-less QR yields
643 // [ R1, R2 ]
644 // [ 0, R3 ]
645 Eigen::HouseholderQR<Eigen::Ref<Eigen::MatrixX<Scalar>>> qr(RR); // In-place QR decomposition
646 Xi = RR.bottomRightCorner(Xi.rows(), Xi.cols()).triangularView<Eigen::Upper>();
647 }
648 }
649 }
650 if (verbosity > 1)
651 std::println("WARNING: maximum number of iterations reached");
652 return 1;
653}

786 return 0;
787 }
788
789 // Evaluate cost and gradient for trial step
790 xn = x + p;
791 Scalar fn = costFunc(xn, gn);
792 bool outOfDomain = !std::isfinite(fn) || !gn.allFinite(); // if any nan, -inf or +inf
793 Vector y = gn - g;
794

796 if (outOfDomain)
797 {
798 rho = -1; // Force trust region reduction and reject step
799 }
800 else
801 {
802 Scalar dm = -pg - 0.5*z.squaredNorm(); // Predicted reduction f - fp, where fp = f + p'*g + 0.5*p'*H*p
803 rho = (f - fn)/dm; // Actual reduction divided by predicted reduction
804 }
805
806 if (rho < 0.1)
807 {
808 Delta = 0.25*z.norm(); // Decrease trust region radius
809 }
810 else
811 {
812 if (rho > 0.75 && z.norm() > 0.8*Delta)
813 {
814 Delta = 2.0*Delta; // Increase trust region radius
815 }
816 }
817
818 if (rho >= 0.001)
819 {
820 // Accept the step
821 x = xn;
822 f = fn;
823 g = gn;
824 }

827 const Scalar sqrteps = std::sqrt(std::numeric_limits<Scalar>::epsilon());
828 if (!outOfDomain)
829 {
830 Scalar py = p.dot(y);
831 if (py > sqrteps*y.norm()*p.norm())
832 {
833 Eigen::SparseQR<Eigen::SparseMatrix<Scalar>, Eigen::COLAMDOrdering<int>> spqr_solver;
834
835 // Compute column-pivoting QR decomposition: [Xi*Pi.'*p; 0]*Pi1 = Q1*RR1 = [Y1, Z1]*[R1; 0], where Pi1 reduces fill-in of R1
836 Eigen::VectorX<Scalar> XiPiTp_0(Xi.rows() + 1);
837 XiPiTp_0 << Xi*Pi.transpose()*p, 0;
838 spqr_solver.compute(XiPiTp_0.sparseView());
839
840 // [Xi*Pi.'; y.'/realsqrt(py)]
841 Eigen::SparseMatrix<Scalar> XiPiT_yTsqrtpy(Xi.rows() + 1, Xi.cols());
842 // Copy Xi*Pi.transpose() into top rows of XiPiT_yTsqrtpy
843 Eigen::SparseMatrix<Scalar> XiPiT = Xi*Pi.transpose();

845 for (typename Eigen::SparseMatrix<Scalar>::InnerIterator it(XiPiT, k); it; ++it)
846 XiPiT_yTsqrtpy.insert(it.row(), it.col()) = it.value();
847 // Copy y.transpose()/std::sqrt(py) into bottom row of XiPiT_yTsqrtpy
848 Eigen::RowVectorX<Scalar> yTsqrtpy = y.transpose()/std::sqrt(py);
849 for (int c = 0; c < Xi.cols(); ++c)
850 XiPiT_yTsqrtpy.insert(Xi.rows(), c) = yTsqrtpy(c);
851
852 // Z1.'*[Xi*Pi.'; y.'/realsqrt(py)]
853 Eigen::SparseMatrix<Scalar> Z1T_XiPiT_yTsqrtpy = (spqr_solver.matrixQ().transpose()*XiPiT_yTsqrtpy.toDense()).bottomRows(Xi.rows()).sparseView();
854
855 // Compute column-pivoting QR decomposition: Z1.'*[Xi*Pi.'; y.'/realsqrt(py)]*Pi2 = Q2*R2, where Pi2 reduces fill-in of R2
856 Z1T_XiPiT_yTsqrtpy.makeCompressed();
857 spqr_solver.compute(Z1T_XiPiT_yTsqrtpy);
858
859 Xi = spqr_solver.matrixR();
860 Pi = spqr_solver.colsPermutation();
861 }
862 }
863 }
864 if (verbosity > 1)
865 std::println("WARNING: maximum number of iterations reached");
866 return 1;
867}

964template <typename Func>
965int BFGSTrustSqrtInv(Func costFunc, Eigen::VectorXd & x, Eigen::VectorXd & g, Eigen::MatrixXd & S, int verbosity = 0)
966{
967 typedef double Scalar;
968 typedef Eigen::VectorXd Vector;
969 typedef Eigen::MatrixXd Matrix;
970
971 assert(x.cols() == 1);
972 g.resize(x.size());
973 assert(S.isUpperTriangular());
974
975 // Storage for trial point and gradient
976 Vector xn(x.size());
977 Vector gn(x.size());
978
979 // Evaluate initial cost and gradient
980 Scalar f = costFunc(x, g);
981 if (!std::isfinite(f) || !g.allFinite()) // if any nan, -inf or +inf
982 {
983 if (verbosity > 1)
984 std::println("ERROR: Initial point is not in domain of cost function");
985 return -1;
986 }
987
988 Scalar Delta = 10e0; // Initial trust-region radius
989
990 const int maxIterations = 5000;
991 for (int i = 0; i < maxIterations; ++i)
992 {
993 // Solve trust-region subproblem
994 Vector p;
995 trsSqrtInv(S, g, Delta, p); // minimise 0.5*p.'*inv(S.'*S)*p + g.'*p subject to ||inv(S.')*p|| <= Delta
996 // Solve S.'*z = p for z
997 Vector z = S.triangularView<Eigen::Upper>().transpose().solve(p);

999 Scalar pg = p.dot(g);

1016 xn = x + p;

1018 bool outOfDomain = !std::isfinite(fn) || !gn.allFinite(); // if any nan, -inf or +inf
1019 Vector y = gn - g;

1021 Scalar rho;
1022 if (outOfDomain)
1023 {
1024 rho = -1; // Force trust region reduction and reject step
1025 }
1026 else
1027 {
1028 Scalar dm = -pg - 0.5*z.squaredNorm(); // Predicted reduction f - fp, where fp = f + p'*g + 0.5*p'*H*p
1029 rho = (f - fn)/dm; // Actual reduction divided by predicted reduction
1030 }

1035 }
1036 else

1040 Delta = 2.0*Delta; // Increase trust region radius
1041 }
1042 }
1043
1044 if (rho >= 0.001)
1045 {
1046 // Accept the step
1047 x = xn;
1048 f = fn;
1049 g = gn;
1050 }
1051
1052 // Update Hessian approximation
1053 const Scalar sqrteps = std::sqrt(std::numeric_limits<Scalar>::epsilon());
1054 if (!outOfDomain)
1055 {
1056 Scalar py = p.dot(y);
1057 if (py > sqrteps*y.norm()*p.norm())
1058 {
1059 // Form
1060 // [ p^T/sqrt(y^T*p) ]
1061 // [ S*(I - y*p^T/(y^T*p)) ]
1062 Matrix RR(x.size() + 1, x.size());

1064 S*(Eigen::MatrixXd::Identity(x.size(), x.size()) - y*p.transpose()/py);

1066 Eigen::HouseholderQR<Eigen::Ref<Eigen::MatrixX<Scalar>>> qr(RR); // In-place QR decomposition
1067 S = RR.topRows(x.size()).triangularView<Eigen::Upper>();
1068 }
1069 }
1070 }
1071 if (verbosity > 1)
1072 std::println("WARNING: maximum number of iterations reached");
1073 return 1;
1074}
