88 Eigen::VectorX<Scalar> simulate() const
89 {
90 static boost::random::mt19937 rng(std::time(0)); // Initialise and seed once
91 boost::random::normal_distribution<> dist;
92
93 // Draw w ~ N(0, I)
94 Eigen::VectorX<Scalar> w(dim());
95 for (Eigen::Index i = 0; i < dim(); ++i)
96 {
97 w(i) = dist(rng);
98 }
99
100 return mean() + sqrtCov().transpose()*w;
101 }

109 static double chi2inv(double p, double nu)
110 {
111 assert(p >= 0);
112 assert(p < 1);
113 assert(nu > 0.0); // added
114 // TODO
115 // Handle a couple of trivial edges
116 if (p == 0.0) return 0.0; // F^{-1}(0)=0 for chi-square
117
118 const double a = 0.5 * nu; // shape for the gamma
119 // Invert the regularized lower incomplete gamma:
120 // P(a, x) = p => x = gamma_p_inv(a, p)
121 const double x = boost::math::gamma_p_inv(a, p);
122
123 return 2.0 * x; // chi-square quantile
124 }

164 static double erfcx(double x)
165 {
166 // https://stackoverflow.com/a/39777361
167 double a, d, e, m, p, q, r, s, t;
168
169 a = std::fmax(x, 0.0 - x); // NaN preserving absolute value computation
170
171 /* Compute q = (a-4)/(a+4) accurately. [0,INF) -> [-1,1] */
172 m = a - 4.0;
173 p = a + 4.0;
174 r = 1.0 / p;
175 q = m * r;
176 t = std::fma(q + 1.0, -4.0, a);
177 e = std::fma(q, -a, t);
178 q = std::fma(r, e, q);
179
180 /* Approximate (1+2*a)*exp(a*a)*erfc(a) as p(q)+1 for q in [-1,1] */
181 p = 0x1.edcad78fc8044p-31; // 8.9820305531190140e-10
182 p = std::fma(p, q, 0x1.b1548f14735d1p-30); // 1.5764464777959401e-09
183 p = std::fma(p, q, -0x1.a1ad2e6c4a7a8p-27); // -1.2155985739342269e-08
184 p = std::fma(p, q, -0x1.1985b48f08574p-26); // -1.6386753783877791e-08
185 p = std::fma(p, q, 0x1.c6a8093ac4f83p-24); // 1.0585794011876720e-07
186 p = std::fma(p, q, 0x1.31c2b2b44b731p-24); // 7.1190423171700940e-08
187 p = std::fma(p, q, -0x1.b87373facb29fp-21); // -8.2040389712752056e-07
188 p = std::fma(p, q, 0x1.3fef1358803b7p-22); // 2.9796165315625938e-07
189 p = std::fma(p, q, 0x1.7eec072bb0be3p-18); // 5.7059822144459833e-06
190 p = std::fma(p, q, -0x1.78a680a741c4ap-17); // -1.1225056665965572e-05
191 p = std::fma(p, q, -0x1.9951f39295cf4p-16); // -2.4397380523258482e-05
192 p = std::fma(p, q, 0x1.3be1255ce180bp-13); // 1.5062307184282616e-04
193 p = std::fma(p, q, -0x1.a1df71176b791p-13); // -1.9925728768782324e-04
194 p = std::fma(p, q, -0x1.8d4aaa0099bc8p-11); // -7.5777369791018515e-04
195 p = std::fma(p, q, 0x1.49c673066c831p-8); // 5.0319701025945277e-03
196 p = std::fma(p, q, -0x1.0962386ea02b7p-6); // -1.6197733983519948e-02
197 p = std::fma(p, q, 0x1.3079edf465cc3p-5); // 3.7167515521269866e-02
198 p = std::fma(p, q, -0x1.0fb06dfedc4ccp-4); // -6.6330365820039094e-02
199 p = std::fma(p, q, 0x1.7fee004e266dfp-4); // 9.3732834999538536e-02
200 p = std::fma(p, q, -0x1.9ddb23c3e14d2p-4); // -1.0103906603588378e-01
201 p = std::fma(p, q, 0x1.16ecefcfa4865p-4); // 6.8097054254651804e-02
202 p = std::fma(p, q, 0x1.f7f5df66fc349p-7); // 1.5379652102610957e-02
203 p = std::fma(p, q, -0x1.1df1ad154a27fp-3); // -1.3962111684056208e-01
204 p = std::fma(p, q, 0x1.dd2c8b74febf6p-3); // 2.3299511862555250e-01
205
206 /* Divide (1+p) by (1+2*a) ==> exp(a*a)*erfc(a) */
207 d = a + 0.5;
208 r = 1.0 / d;
209 r = r * 0.5;
210 q = std::fma(p, r, r); // q = (p+1)/(1+2*a)
211 t = q + q;
212 e = (p - q) + std::fma(t, -a, 1.0); // residual: (p+1)-q*(1+2*a)
213 r = std::fma(e, r, q);
214
215 /* Handle argument of infinity */
216 if (a > 0x1.fffffffffffffp1023) r = 0.0;
217
218 /* Handle negative arguments: erfcx(x) = 2*exp(x*x) - erfcx(|x|) */
219 if (x < 0.0)
220 {
221 s = x * x;
222 d = std::fma(x, x, -s);
223 e = std::exp(s);
224 r = e - r;
225 r = std::fma(e, d + d, r);
226 r = r + e;
227 if (e > 0x1.fffffffffffffp1023) r = e; // avoid creating NaN
228 }
229 return r;
230 }

250 Eigen::Matrix<Scalar, 2, Eigen::Dynamic> confidenceEllipse(double nSigma = 3.0, int nSamples = 100) const
251 {
252 const Eigen::Index & n = dim();
253 assert(n == 2);
254 assert(nSamples >= 2); // added
255
256 Eigen::Matrix<Scalar, 2, Eigen::Dynamic> X(2, nSamples);
257
258 // TODO
259 // Probability that corresponds to "± nSigma" in 1D
260 const double p = 2.0 * normcdf(static_cast<double>(nSigma)) - 1.0;
261 // Chi-square threshold for 2 DoF, then radius in whitened coordinates
262 const double thr = chi2inv(p, 2.0);
263 const Scalar r = static_cast<Scalar>(std::sqrt(thr));
264
265 const Eigen::MatrixX<Scalar> ST = sqrtCov().transpose(); // S^T
266
267 // Sample angles; make last point equal to the first to close the loop
268 for (int i = 0; i < nSamples; ++i)
269 {
270 const double t =
271 (nSamples == 1) ? 0.0 : (2.0 * std::numbers::pi * i) / static_cast<double>(nSamples - 1);
272
273 const Scalar c = static_cast<Scalar>(std::cos(t));
274 const Scalar s = static_cast<Scalar>(std::sin(t));
275
276 // y = r * [cos t; sin t]
277 Eigen::Matrix<Scalar, 2, 1> y;
278 y << r * c, r * s;
279
280 // x = mu + S^T * y
281 X.col(i) = mean() + ST * y;
282 }
283
284 assert(X.cols() == nSamples);
285 assert(X.rows() == 2);
286 return X;
287 }

332 Scalar logIntegral(Scalar a, Scalar b, Scalar & dldmu) const
333 {
334 Scalar za, zb, ta, tb;
335 Scalar l = logIntegralImpl(a, b, za, zb, ta, tb);
336 Scalar sigma = sqrtCov()(0, 0);
337 using std::exp, std::log;
338 dldmu = exp(-za * za - l - 0.5 * log(2 * std::numbers::pi * sigma * sigma)) -
339 exp(-zb * zb - l - 0.5 * log(2 * std::numbers::pi * sigma * sigma));
340
341 return l;
342 }

344private:
355 Scalar logIntegralImpl(Scalar a, Scalar b, Scalar & za, Scalar & zb, Scalar & ta, Scalar & tb) const
356 {
357 assert(dim() == 1 && "Expected univariate Gaussian to compute the log-integral");
358 assert(b >= a && "Expected integral to be non-negative so that the log exists");
359
360 using std::abs, std::exp, std::log, std::log1p;
361
362 Scalar mu = mean()(0);
363 Scalar sigma = sqrtCov()(0, 0);
364
365 za = -(std::numbers::sqrt2 / 2.0) * (a - mu) / sigma;
366 zb = -(std::numbers::sqrt2 / 2.0) * (b - mu) / sigma;
367 ta = log(0.5 * erfcx(abs(za))) - za * za;
368 tb = log(0.5 * erfcx(abs(zb))) - zb * zb;
369
370 if (za >= 0)
371 {
372 if (zb >= 0)
373 {
374 return tb + log1p(-exp(ta - tb));
375 }
376 else // zb < 0
377 {
378 return log1p(-exp(ta) - exp(tb));
379 }
380 }
381 else // za < 0
382 {
383 assert(zb < 0 && "Case forbidden, since b >= a implies 0 <= zb <= za < 0");
384 return ta + log1p(-exp(tb - ta));
385 }
386 }
