144 static double erfcx(double x)
145 {
146 // https://stackoverflow.com/a/39777361
147 double a, d, e, m, p, q, r, s, t;
148
149 a = std::fmax(x, 0.0 - x); // NaN preserving absolute value computation
150
151 /* Compute q = (a-4)/(a+4) accurately. [0,INF) -> [-1,1] */
152 m = a - 4.0;
153 p = a + 4.0;
154 r = 1.0 / p;
155 q = m * r;
156 t = std::fma(q + 1.0, -4.0, a);
157 e = std::fma(q, -a, t);
158 q = std::fma(r, e, q);
159
160 /* Approximate (1+2*a)*exp(a*a)*erfc(a) as p(q)+1 for q in [-1,1] */
161 p = 0x1.edcad78fc8044p-31; // 8.9820305531190140e-10

181 p = std::fma(p, q, 0x1.16ecefcfa4865p-4); // 6.8097054254651804e-02
182 p = std::fma(p, q, 0x1.f7f5df66fc349p-7); // 1.5379652102610957e-02
183 p = std::fma(p, q, -0x1.1df1ad154a27fp-3); // -1.3962111684056208e-01
184 p = std::fma(p, q, 0x1.dd2c8b74febf6p-3); // 2.3299511862555250e-01
185
186 /* Divide (1+p) by (1+2*a) ==> exp(a*a)*erfc(a) */
187 d = a + 0.5;
188 r = 1.0 / d;
189 r = r * 0.5;
190 q = std::fma(p, r, r); // q = (p+1)/(1+2*a)
191 t = q + q;
192 e = (p - q) + std::fma(t, -a, 1.0); // residual: (p+1)-q*(1+2*a)
193 r = std::fma(e, r, q);
194
195 /* Handle argument of infinity */
196 if (a > 0x1.fffffffffffffp1023) r = 0.0;
197
198 /* Handle negative arguments: erfcx(x) = 2*exp(x*x) - erfcx(|x|) */
199 if (x < 0.0)
200 {
201 s = x * x;
202 d = std::fma(x, x, -s);
203 e = std::exp(s);
204 r = e - r;
205 r = std::fma(e, d + d, r);
206 r = r + e;
207 if (e > 0x1.fffffffffffffp1023) r = e; // avoid creating NaN
208 }
209 return r;
210 }

285 Scalar logIntegral(Scalar a, Scalar b, Scalar & dldmu) const
286 {
287 Scalar za, zb, ta, tb;
288 Scalar l = logIntegralImpl(a, b, za, zb, ta, tb);
289 Scalar sigma = sqrtCov()(0, 0);
290 using std::exp, std::log;
291 dldmu = exp(-za * za - l - 0.5 * log(2 * std::numbers::pi * sigma * sigma)) -
292 exp(-zb * zb - l - 0.5 * log(2 * std::numbers::pi * sigma * sigma));
293
294 return l;
295 }

308 Scalar logIntegralImpl(Scalar a, Scalar b, Scalar & za, Scalar & zb, Scalar & ta, Scalar & tb) const
309 {
310 assert(dim() == 1 && "Expected univariate Gaussian to compute the log-integral");
311 assert(b >= a && "Expected integral to be non-negative so that the log exists");
312
313 using std::abs, std::exp, std::log, std::log1p;
314
315 Scalar mu = mean()(0);
316 Scalar sigma = sqrtCov()(0, 0);
317
318 za = -(std::numbers::sqrt2 / 2.0) * (a - mu) / sigma;
319 zb = -(std::numbers::sqrt2 / 2.0) * (b - mu) / sigma;
320 ta = log(0.5 * erfcx(abs(za))) - za * za;
321 tb = log(0.5 * erfcx(abs(zb))) - zb * zb;
322
323 if (za >= 0)
324 {
325 if (zb >= 0)
326 {
327 return tb + log1p(-exp(ta - tb));
328 }
329 else // zb < 0
330 {
331 return log1p(-exp(ta) - exp(tb));
332 }
333 }
334 else // za < 0
335 {
336 assert(zb < 0 && "Case forbidden, since b >= a implies 0 <= zb <= za < 0");
337 return ta + log1p(-exp(tb - ta));
338 }
339 }
