216 {

446 static Gaussian fromSamples(const Eigen::MatrixX<Scalar> & X)
447 {
448 const Eigen::Index n = X.rows();
449 const Eigen::Index m = X.cols();
450
451 Gaussian out;

453 // Compute the sample mean
454 out.mu_ = X.rowwise().mean();

459 out.S_ = SS.topRows(n).template triangularView<Eigen::Upper>();

519 // S_(idxB, idxA).transpose()*
520 // S_(idxB, idxB).eval().template triangularView<Eigen::Upper>().transpose().solve(xB - mu_(idxB));
521 // out.S_ = S_(idxA, idxA);

527 const Eigen::Index nA = static_cast<Eigen::Index>(idxA.size());
528 const Eigen::Index nB = static_cast<Eigen::Index>(idxB.size());
529 const Eigen::Index n = dim();
530 assert(nA + nB == n);
531 assert(xB.size() == nB);

533 // (1) Build the column concatenation [ S_B S_A ] = [ S(:, idxB) S(:, idxA) ]
534 Eigen::Matrix<Scalar, Eigen::Dynamic, Eigen::Dynamic> SS(n, nB + nA);
535 SS.leftCols(nB) = S_(Eigen::all, idxB);
536 SS.rightCols(nA) = S_(Eigen::all, idxA);
537
538 // (2) Q-less QR on [ S_B S_A ]
539 Eigen::HouseholderQR<Eigen::Ref<Eigen::Matrix<Scalar, Eigen::Dynamic, Eigen::Dynamic>>> qr(SS);
540
541 // Extract the square R = [ S1 S2 ; 0 S3 ] from the top rows and keep only the upper triangle.
542 Eigen::Matrix<Scalar, Eigen::Dynamic, Eigen::Dynamic> R =
543 SS.topRows(nB + nA).template triangularView<Eigen::Upper>();
544
545 // (3) Partition R into its blocks:
546 // R = [ R1 R2 ;
547 // 0 R3 ]
548 const auto R1 = R.topLeftCorner(nB, nB); // upper-triangular
549 const auto R2 = R.topRightCorner(nB, nA); //(nBnA)
550 const auto R3 = R.bottomRightCorner(nA, nA); // nAnA, upper triangular
551
552 // 4) Conditional mean: _A|B = _A + (R1^{-1} R2)^T (xB - _B)
553 // Compute K = R1 \ R2 via triangular solve, then K^T * deltaB (lab formula)
554 Eigen::Matrix<Scalar, Eigen::Dynamic, Eigen::Dynamic> K = R1.template triangularView<Eigen::Upper>().solve(R2); // nBnA
555 out.mu_ = mu_(idxA) + K.transpose() * (xB - mu_(idxB));
556
557 // 5) Conditional sqrt covariance is R3 directly S_{A|B} = S3
558 out.S_ = R3;
559
560 return out;
561 }

950 Eigen::Matrix4<Scalar> quadricSurface(double nSigma = 3.0) const
951 {
952 const Eigen::Index & n = dim();
953 assert(n == 3);
954
955 Eigen::Matrix4<Scalar> Q;
956 // TODO
957 // A = P^{-1} using the stable infoMat() (no explicit inversion)
958 const Eigen::MatrixX<Scalar> A = infoMat(); // 3x3
959 const Eigen::VectorX<Scalar> mu = mean(); // 3x1
960 const Eigen::VectorX<Scalar> k = A * mu; // 3x1
961
962 // Probability that corresponds to " nSigma" in 1D
963 const double p = 2.0 * normcdf(static_cast<double>(nSigma)) - 1.0;
964 // Chi-square threshold for 3 DoF
965 const double c3 = chi2inv(p, 3.0);
966
967 Q.setZero();
968 Q.template topLeftCorner<3,3>() = A;
969 Q.template topRightCorner<3,1>() = -k;
970 Q.template bottomLeftCorner<1,3>() = -k.transpose();
971 Q(3,3) = static_cast<Scalar>(mu.dot(k) - c3);
972
973 return Q;
974 }
