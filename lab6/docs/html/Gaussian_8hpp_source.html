215 static Gaussian fromSqrtInfo(const Eigen::VectorX<Scalar> & nu, const Eigen::MatrixX<Scalar> & Xi)
216 {
217 assert(nu.size() == Xi.cols());
218 assert(Xi.isUpperTriangular());
219 Gaussian out;
220 // Solve Xi*mu = nu
221 out.mu_ = Xi.template triangularView<Eigen::Upper>().solve(nu);
222 // S = qr(Xi^{-T})
223 out.S_ = Xi.template triangularView<Eigen::Upper>().transpose().solve(
224 Eigen::MatrixX<Scalar>::Identity(Xi.cols(), Xi.cols())
225 );
226 Eigen::HouseholderQR<Eigen::Ref<Eigen::MatrixX<Scalar>>> qr(out.S_); // In-place QR decomposition
227 out.S_ = out.S_.template triangularView<Eigen::Upper>(); // Safe aliasing
228 return out;
229 }

245 static Gaussian fromInfo(const Eigen::VectorX<Scalar> & eta, const Eigen::MatrixX<Scalar> & Lambda)
246 {
247 assert(eta.size() == Lambda.cols());
248 assert(Lambda.rows() == Lambda.cols());
249 // Let Xi be an upper-triangular matrix such that Xi^T*Xi = Lambda
250 Eigen::LLT<Eigen::MatrixX<Scalar>, Eigen::Upper> llt(Lambda);
251 Eigen::MatrixX<Scalar> Xi = llt.matrixU();

253 Eigen::VectorX<Scalar> nu = Xi.template triangularView<Eigen::Upper>().transpose().solve(eta);

446 static Gaussian fromSamples(const Eigen::MatrixX<Scalar> & X)
447 {
448 const Eigen::Index n = X.rows();
449 const Eigen::Index m = X.cols();
450
451 Gaussian out;
452
453 // Compute the sample mean
454 out.mu_ = X.rowwise().mean();
455
456 // Compute the sample square-root covariance
457 Eigen::MatrixX<Scalar> SS = std::sqrt(1.0 / (m - 1))*(X.colwise() - out.mu_).transpose();
458 Eigen::HouseholderQR<Eigen::Ref<Eigen::MatrixX<Scalar>>> qr(SS); // In-place QR decomposition
459 out.S_ = SS.topRows(n).template triangularView<Eigen::Upper>();
460 return out;
461 }

525 template <typename IndexTypeA, typename IndexTypeB>
526 Gaussian conditional(const IndexTypeA & idxA, const IndexTypeB & idxB, const Gaussian & pxB_y) const
527 {
528 const std::size_t & nA = idxA.size();
529 const std::size_t & nB = idxB.size();
530
531 // Form [S(:, idxB), S(:, idxA)]
532 Eigen::MatrixX<Scalar> SS(dim(), nA + nB);
533 SS << S_(Eigen::all, idxB), S_(Eigen::all, idxA);
534 // Q-less QR yields
535 // [S1, S2;
536 // 0, S3]
537 Eigen::HouseholderQR<Eigen::Ref<Eigen::MatrixX<Scalar>>> qr(SS); // In-place QR decomposition
538
539 // K = S1\S2
540 Eigen::MatrixX<Scalar> K = SS.topLeftCorner(nB, nB).template triangularView<Eigen::Upper>().solve(SS.topRightCorner(nB, nA));
541 Eigen::MatrixX<Scalar> S3 = SS.bottomRightCorner(nA, nA).template triangularView<Eigen::Upper>();
542
543 // Form [pxB_y.S*K; S3]
544 Eigen::MatrixX<Scalar> RR(nA + nB, nA);
545 RR << pxB_y.S_ * K,
546 S3;
547 // Q-less QR yields
548 // [R1;
549 // 0]
550 Eigen::HouseholderQR<Eigen::Ref<Eigen::MatrixX<Scalar>>> qr_RR(RR);
551
552 // p(x(idxA) | y) = N^-0.5(x(idxA); mu(idxA) + K.'*(pxB_y.mu - mu(idxB))), R1)
553 Gaussian out;
554 out.mu_ = mu_(idxA) + K.transpose()*(pxB_y.mean() - mu_(idxB));
555 out.S_ = RR.topRows(nA).template triangularView<Eigen::Upper>();
556 return out;
557 }

596 assert(x.cols() == 1);
597 assert(x.size() == dim());
598
599 // Compute log N(x; mu, P) where P = S.'*S
600 // log N(x; mu, P) = -0.5*(x - mu).'*inv(P)*(x - mu) - 0.5*log(det(2*pi*P))
601
602 // TODO: Numerically stable version
603 const Eigen::Index n = dim();
604 const Eigen::VectorX<Scalar> e = x - mu_;
605
606 // y = S^{-T} e (solve S^T y = e), so e^T P^{-1} e = ||y||^2
607 const Eigen::VectorX<Scalar> y =
608 S_.template triangularView<Eigen::Upper>().transpose().solve(e);
609 const Scalar quad = y.squaredNorm();
610
611 using std::log;
612 using std::abs;
613
614 // 0.5 * log|P| = sum_i log(|S_ii|)
615 Scalar half_logdetP = Scalar(0);
616 for (Eigen::Index i = 0; i < n; ++i) {
617 half_logdetP += log(abs(S_(i, i)));
618 }
619
620 const Scalar log2pi = log(Scalar(2) * Scalar(std::numbers::pi));
621 // log N(x; μ, P) = -0.5*quad - 0.5*n*log(2π) - 0.5*log|P|
622 // and 0.5*log|P| = half_logdetP
623 return Scalar(-0.5) * quad - Scalar(0.5) * Scalar(n) * log2pi - half_logdetP;
