98// Templated prediction function (the core logic)
99// This must be in the header file to be available for instantiation by other files and autodiff.
100template<typename Scalar>
101Eigen::Vector3<Scalar>
102MeasurementSLAMDuckBundle::predictDuckT(const Eigen::VectorX<Scalar>& x,
103 const SystemSLAM& system,
104 std::size_t idxLandmark) const
105{
106 // Get camera pose from the state vector x
107 Pose<Scalar> Tnc;
108 Tnc.translationVector = system.cameraPosition(camera_, x);
109 Tnc.rotationMatrix = system.cameraOrientation(camera_, x);
110
111 // Get the landmark's 3D position from the state vector x
112 const std::size_t idx = system.landmarkPositionIndex(idxLandmark);
113 Eigen::Vector3<Scalar> rLNn = x.template segment<3>(idx);
114
115 // Transform the landmark's position from world coordinates to camera coordinates
116 const Eigen::Matrix3<Scalar> Rcn = Tnc.rotationMatrix.transpose();
117 const Eigen::Vector3<Scalar> rLCc = Rcn * (rLNn - Tnc.translationVector);
118
119 // Project the 3D point in camera coordinates to 2D pixel coordinates
120 const Eigen::Vector2<Scalar> uv = camera_.vectorToPixel(rLCc);
121
122 // Calculate the predicted area based on the landmark's depth
123 const Scalar depth = rLCc.norm();
124 const Scalar A = (Scalar(fx_) * Scalar(fy_) * Scalar(std::numbers::pi) * Scalar(duck_r_m_) * Scalar(duck_r_m_)) / (depth * depth);
125
126 // Assemble the 3D measurement vector [u, v, Area]
127 Eigen::Vector3<Scalar> h;
128 h << uv(0), uv(1), A;
129 return h;
130}
