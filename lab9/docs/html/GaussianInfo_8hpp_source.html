131 static GaussianInfo fromSqrtMoment(const Eigen::VectorX<Scalar> & mu, const Eigen::MatrixX<Scalar> & S)
132 {
133 assert(mu.size() == S.cols());
134 assert(S.isUpperTriangular());
135
136 GaussianInfo out(S.cols());
137
138 // qr(S^{-T})
139 out.Xi_ = S.template triangularView<Eigen::Upper>().transpose().solve(
140 Eigen::MatrixX<Scalar>::Identity(S.cols(), S.cols())
141 );
142 Eigen::HouseholderQR<Eigen::Ref<Eigen::MatrixX<Scalar>>> qr(out.Xi_); // In-place QR decomposition
143 out.Xi_ = out.Xi_.template triangularView<Eigen::Upper>(); // Safe aliasing
144
145 // Xi*mu = nu
146 out.nu_ = out.Xi_*mu;
147
148 return out;
149 }

389 const Eigen::Index n = X.rows();

398 Eigen::MatrixXd S = SS.topRows(n).template triangularView<Eigen::Upper>();
399 return GaussianInfo::fromSqrtMoment(mu, S);

585 template <typename Func>
586 GaussianInfo affineTransform(Func h) const
587 {
588 Eigen::MatrixX<Scalar> J;
589 Eigen::VectorX<Scalar> mux = mean();
590 Eigen::VectorX<Scalar> muy = h(mux, J); // Evaluate function at mean value
591 const std::size_t m = J.rows();
592 const std::size_t n = J.cols();
593 assert(m == muy.size());
594 assert(n == dim());
595 // assert(n == static_cast<Eigen::Index>(dim()));
596
597 // Linearise y = h(x) about x = x y h(x) + J (x x) = J x + b
598 const Eigen::VectorX<Scalar> b = muy - J * mux; // helper
599
600 // MATLAB
601 // [U, s, V] = svd(J, "vector"); % J = U * diag(s) * V.'
602 // tol = max(m, n)*eps(max(s));
603 // r = nnz(s > tol);
604 // U1 = U(:,1:r); U2 = U(:,r+1:m);
605 // V1 = V(:,1:r); V2 = V(:,r+1:n);
606 // s1 = s(1:r);
607
608 // --- SVD of J (FULL factors to allow U2/V2 even when rank-deficient) ---
609 Eigen::JacobiSVD<Eigen::MatrixX<Scalar>> svd(J, Eigen::ComputeFullU | Eigen::ComputeFullV);
610 const Eigen::VectorX<Scalar> s = svd.singularValues(); // length = min(m,n)
611 const Eigen::MatrixX<Scalar> U = svd.matrixU(); // m x m
612 const Eigen::MatrixX<Scalar> V = svd.matrixV(); // n x n
613
614 // call svd.rank TODO
615
616 // Numerical rank threshold: tol = max(m,n) * eps * max(s)
617 const Scalar tol = std::max(m, n) * Eigen::NumTraits<Scalar>::epsilon() * (s.size() ? s.maxCoeff() : Scalar(0));
618 const Eigen::Index r = (s.array() > tol).count(); // MATLAB: r = nnz(s > tol);
619
620 // Partitions matching MATLAB slices
621 const Eigen::MatrixX<Scalar> U1 = U.leftCols(r);
622 Eigen::MatrixX<Scalar> U2; U2.resize(m, m - r); if (m > r) U2 = U.rightCols(m - r);
623 const Eigen::MatrixX<Scalar> V1 = V.leftCols(r);
624 Eigen::MatrixX<Scalar> V2; V2.resize(n, n - r); if (n > r) V2 = V.rightCols(n - r);
625 const Eigen::VectorX<Scalar> s1 = s.head(r);
626
627 // MATLAB
628 // Jp = (V1./s1(:).')*U1.'; % J^+ = V1 * diag(1./s1) * U1.'
629
630 // Pseudoinverse on rank-r subspace: J = V1 * diag(1./s1) * U1
631 Eigen::MatrixX<Scalar> Jp;
632 if (r > 0) {
633 Jp = V1 * s1.cwiseInverse().asDiagonal() * U1.transpose(); // n x m
634 } else {
635 Jp.resize(n, m);
636 Jp.setZero();
637 }
638
639 // MATLAB
640 // X = obj.Xi*V2;
641 // Y = obj.Xi*Jp;
642
643 // Build X = Xi*V2 (nullspace part) and Y = Xi*Jp (range part)
644 Eigen::MatrixX<Scalar> X; X.resize(n, n - r);
645 if (V2.cols() > 0) X = Xi_ * V2; else X.setZero();
646 Eigen::MatrixX<Scalar> Y = Xi_ * Jp; // n x m
647
648 // MATLAB
649 // sigma_max_ub = realsqrt(sum([X, Y].^2, 'all')); % Cheap upper bound for the largest singular value of [X, Y]
650 // kappa = 1e7*sigma_max_ub; % Constraint information factor (kappa >> sigma_max_ub)
651
652 // Constraint scaling: kappa >> ||[X Y]|| (cheap upper bound via Frobenius norm)
653 const Scalar sigma_max_ub = std::sqrt((X.array().square().sum()) + (Y.array().square().sum()));
654 const Scalar kappa = std::max<Scalar>(Scalar(1e7) * sigma_max_ub, Scalar(1e-3)); // floor to keep constraints active same as MATLAB
655 // const Scalar kappa = (sigma_max_ub > Scalar(0)) ? Scalar(1e7) * sigma_max_ub : Scalar(1);
656
657 // MATLAB
658 // RR = qr([X, Y, obj.nu + Y*b; ...
659 // zeros(m - r, n - r), kappa*U2.', kappa*U2.'*b]);
660 // % Q-less QR yields
661 // % [R1, R2, nu1;
662 // % 0, R3, nu2]
663
664 // C++: assemble the augmented matrix exactly the same way
665 const Eigen::Index rows = (n - r) + m;
666 const Eigen::Index cols = (n - r) + m + 1;
667 Eigen::MatrixX<Scalar> A(rows, cols);
668 A.setZero();
669
670 // Top block (n rows)
671 if (X.cols() > 0) A.block(0, 0, n, n - r) = X;
672 A.block(0, n - r, n, m) = Y;
673 A.block(0, (n - r) + m, n, 1) = nu_ + Y * b;
674
675 // Bottom block ((m - r) rows) only present if rank-deficient (m > r)
676 if (m > r) {
677 // Left block already zeroed
678 A.block(n, n - r, m - r, m) = kappa * U2.transpose();
679 A.block(n, (n - r) + m, m - r, 1) = kappa * (U2.transpose() * b);
680 }
681
682 // Q-less QR (Eigen: obtain R from the packed QR and take its upper-triangular view)
683 Eigen::HouseholderQR<Eigen::MatrixX<Scalar>> qr(A);
684 Eigen::MatrixX<Scalar> R = qr.matrixQR()
685 .topLeftCorner(rows, cols)
686 .template triangularView<Eigen::Upper>();
687
688 // MATLAB
689 // R3 = RR(n-r+1:n-r+m, n-r+1:n-r+m);
690 // nu2 = RR(n-r+1:n-r+m, n-r+m+1);
691
692 // Read off the bottom-right (m x m) block and RHS following the Y-columns region.
693 const Eigen::Index row0 = (n - r); // start row for the y-block
694 const Eigen::Index col0 = (n - r); // first y-column in A (i.e., where Y starts)
695 Eigen::MatrixX<Scalar> R3 = R.block(row0, col0, m, m);
696 Eigen::VectorX<Scalar> nu2 = R.block(row0, col0 + m, m, 1);
697
698 // Return p(y) = N^{-1/2}(y; nu2, R3)
699 GaussianInfo out(m);
700 out.Xi_ = R3; // upper-triangular by construction
701 out.nu_ = nu2;
702 return out;
703
704 // ------------------------------------- template -------------------------------------
705 // // Linearise y = h(x) about x = mux
706 // // y ~= h(mux) + J*(x - mux)
707 // // = J*x + h(mux) - J*mux
708
709 // // Solve Y*J = Xi for Y
710 // // Y = Xi*inv(J) = (inv(J.')*Xi.').'
711 // Eigen::MatrixX<Scalar> Y = J.transpose().lu().solve(Xi_.transpose()).transpose();
712
713 // // Form [Y, nu]
714 // Eigen::MatrixX<Scalar> RR(n, n + 1);
715 // RR << Y, nu_;
716 // // Q-less QR yields
717 // // [R, r]
718 // Eigen::HouseholderQR<Eigen::Ref<Eigen::MatrixX<Scalar>>> qr(RR); // In-place QR decomposition
719
720 // // p(y) = N^-0.5(y; r + R*(h(mux) - J*mux), R)
721 // GaussianInfo out(m);
722 // out.Xi_ = RR.leftCols(m).template triangularView<Eigen::Upper>();
723 // out.nu_ = RR.col(m) + out.Xi_*(muy - J*mux);
724 // return out;
725 // ------------------------------------- template -------------------------------------
726 }
